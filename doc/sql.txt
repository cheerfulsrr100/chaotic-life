create database chaotic_life;

create table if not exists `chaotic_life_user`
(
    `user_id`           int unsigned primary key auto_increment,
    `user_name`         varchar(20) not null,
    `user_phone_number` varchar(11),
    `user_email`        varchar(40)
) engine = innodb
  default charset = utf8mb4;

create table if not exists `chaotic_life_bill_kind`
(
  `bill_kind_id` int unsigned primary key auto_increment,
  `user_id`      int unsigned not null
) engine = innodb
default charset = utf8mb4;
